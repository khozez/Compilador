        SHORT clientes;cuotas,

        {
            **FUNCION LOCAL A METODO
            VOID cliente_menos()
            {
                clientes = clientes - 1_s,
                RETURN,
            }

            VOID cliente_mas()                                      **ERROR: MAS DE 1 FUNCION LOCAL EN METODO
            {
                clientes = clientes + 1_s,
                RETURN,
            }

        },
