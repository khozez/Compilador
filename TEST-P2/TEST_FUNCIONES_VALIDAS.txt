    {
        cuotas = cant,
        RETURN,
    }

    VOID aumentar_cuota()
    {
        cuotas = cuotas + 1_s,
        RETURN,
    }

    VOID aumentar_cuotas(SHORT cant)
    {
        cuotas = cuotas + cant,
        RETURN,
    }

    VOID pagar_cuotas(SHORT cant)
    {
        SHORT deuda,
        deuda = cuotas--,
        IF (cant > 3_s){
            PRINT %CANTIDAD DE DEUDAS MAYOR A 3

            aumentar_cuotas(2_s),
        }ELSE
        {
            PRINT %CANTIDAD DE DEUDAS MENOR A 3
                   SE AUMENTA 1 CUOTAS%,
            aumentar_cuota(),
        }END_IF,
        pagas = pagas + 1_s,
        RETURN,
    }

        SHORT clientes;cuotas,

        {
            **FUNCION LOCAL A METODO

            {
                clientes = clientes - 1_s,

            }

            IF (cant > 4_s)
                cliente_menos(),
            END_IF,
        },
    }

    VOID recursiva()
    {
        SHORT r,
        r = 4_s,

        RETURN,
